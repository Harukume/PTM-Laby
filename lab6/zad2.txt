ljmp start

P5 equ 0F8H
P7 equ 0DBH
	
LCDstatus  equ 0FF2EH       ; adres do odczytu gotowosci LCD
LCDcontrol equ 0FF2CH       ; adres do podania bajtu sterujacego LCD
LCDdataWR  equ 0FF2DH       ; adres do podania kodu ASCII na LCD

// bajty sterujace LCD
#define  HOME     0x80     
#define  INITDISP 0x38     
#define  HOM2     0xc0     
#define  LCDON    0x0e     
#define  CLEAR    0x01     

// definicje linii klawiatury (wiersze sterowane przez P5)
#define LINE_1		0x7f	// 0111 1111
#define LINE_2		0xbf	// 1011 1111
#define	LINE_3		0xdf	// 1101 1111
#define LINE_4		0xef	// 1110 1111

; --- REJESTRY PAMIĘCI RAM (ZMIENNE POMOCNICZE) ---
LIMIT_WPR   EQU 30H   ; Przechowuje aktualny limit (24 lub 60)
BUFOR_CYFR  EQU 31H   ; Przechowuje aktualnie budowaną liczbę dwucyfrową
ZAPIS_KLAW  EQU 32H   ; Przechowuje zdekodowany kod klawisza

ORG 000BH     				; obsluga przerwania Timera 0
	MOV TH0, #3CH 			; przeladowanie na 50ms
	MOV TL0, #0B0H 			
	DEC R0        			; licznik programowy (20 * 50ms = 1 sekunda)
	RETI          			

org 0100H
		
// makra do obsługi wyświetlacza LCD
LCDcntrlWR MACRO x          
           LOCAL loop       
loop: MOV  DPTR,#LCDstatus  
      MOVX A,@DPTR          
      JB   ACC.7,loop       
      MOV  DPTR,#LCDcontrol 
      MOV  A, x             
      MOVX @DPTR,A          
      ENDM
	  
LCDcharWR MACRO
      LOCAL tutu            
      PUSH ACC              
tutu: MOV  DPTR,#LCDstatus  
      MOVX A,@DPTR          
      JB   ACC.7,tutu       
      MOV  DPTR,#LCDdataWR  
      POP  ACC              
      MOVX @DPTR,A          
      ENDM
	  
init_LCD MACRO
         LCDcntrlWR #INITDISP 
         LCDcntrlWR #CLEAR    
         LCDcntrlWR #LCDON    
         ENDM
		 
// wypisanie liczby dwucyfrowej na LCD (format 00-59)
putdigitLCD:	mov b, #10
				div ab				
				add a, #30H			
				acall putcharLCD
				mov a, b			
				add a, #30H			
				acall putcharLCD
				ret

putcharLCD:	LCDcharWR
			ret

; odświeżenie wyświetlacza zegara
POKAZ_ZEGAR:
	LCDcntrlWR #HOME	
	MOV A,R5		; godziny
	ACALL putdigitLCD
	MOV A,#":"		
	ACALL putcharLCD
	MOV A,R6		; minuty
	ACALL putdigitLCD
	MOV A,#":"		
	ACALL putcharLCD
	MOV A,R7		; sekundy
	ACALL putdigitLCD
	RET

; aktualizacja czasu w tle (wywoływana co 1s)
ZEGAR:
	INC R7		
	MOV A,R7
	CLR C
	SUBB A,#60	
	JZ MINUTY
	ACALL POKAZ_ZEGAR
	RET

MINUTY:
	MOV R7,#00H	
	INC R6		
	MOV A,R6
	CLR C
	SUBB A,#60	
	JZ GODZINY
	ACALL POKAZ_ZEGAR
	RET

GODZINY:
	MOV R6,#00H	
	INC R5		
	MOV A,R5
	CLR C
	SUBB A,#24	
	JNZ EKRAN
	MOV R5,#00H	
EKRAN:
	ACALL POKAZ_ZEGAR
	RET

; manualna korekta przyciskiem PLUS
GODZINA_PLUS:
	INC R5		
	MOV A,R5
	CLR C
	SUBB A,#24	
	JNZ PLUS_OK
	MOV R5,#00H	
PLUS_OK:
	ACALL POKAZ_ZEGAR
	RET

; manualna korekta przyciskiem MINUS
GODZINA_MINUS:
	MOV A,R5
	JNZ MINUS_NORMAL
	MOV R5,#23	
	SJMP MINUS_END
MINUS_NORMAL:
	DEC R5		
MINUS_END:
	ACALL POKAZ_ZEGAR
	RET

; =========================================================================
; SPRZĘTOWY ODCZYT I DEKODOWANIE KLAWIATURY MATRYCOWEJ
; =========================================================================

CZYTAJ_KLAWISZ:
	; Sprawdzanie Linii 1
	MOV P5,#LINE_1
	MOV A,P7
	ANL A,#0FH
	CJNE A,#0FH, KLAW_ZNALEZIONY
	
	; Sprawdzanie Linii 2
	MOV P5,#LINE_2
	MOV A,P7
	ANL A,#0FH
	CJNE A,#0FH, KLAW_ZNALEZIONY
	
	; Sprawdzanie Linii 3
	MOV P5,#LINE_3
	MOV A,P7
	ANL A,#0FH
	CJNE A,#0FH, KLAW_ZNALEZIONY
	
	; Sprawdzanie Linii 4
	MOV P5,#LINE_4
	MOV A,P7
	ANL A,#0FH
	CJNE A,#0FH, KLAW_ZNALEZIONY
	
	MOV A,#0FFH			; brak wciśniętego klawisza
	RET

KLAW_ZNALEZIONY:
	; Budowanie unikalnego kodu klawisza: kombinacja wiersza i kolumny
	MOV R1, A			; zachowaj stan kolumn (P7) w R1
	MOV A, P5			; pobierz aktualnie aktywny wiersz
	ANL A, #0F0H		; maskuj starsze bity wiersza
	ORL A, R1			; połącz wiersz z kolumną
	MOV R1, A			; ostateczny unikalny kod w R1

	; Mapowanie kodu z tabeli znaków
	MOV DPTR,#TAB_KLAWISZ
	MOV R2,#0
SZUKAJ_KODU:
	MOV A,R2
	MOVC A,@A+DPTR
	XRL A,R1
	JZ ZNALEZIONO
	INC R2
	CJNE R2,#12,SZUKAJ_KODU
	MOV A,#0FFH
	RET
ZNALEZIONO:
	MOV A,R2			; zwraca pozycję (0-9 = cyfry, 10 = *, 11 = #)
	RET

CZEKAJ_NA_KLAWISZ:
	; Krok 1: Czekaj tak długo, aż wykryjesz wciśnięcie klawisza
	ACALL CZYTAJ_KLAWISZ
	XRL A,#0FFH
	JZ CZEKAJ_NA_KLAWISZ
	
	MOV ZAPIS_KLAW, A	; Zapamiętaj wykryty klawisz
	
	; Krok 2: Czekaj tak długo, aż użytkownik CAŁKOWICIE PUŚCI klawisz
CZEKAJ_PUSZCZENIE:
	MOV P5, #00H		; Zwolnij linie sterujące klawiatury
	MOV A, P7
	ANL A, #0FH
	CJNE A, #0FH, CZEKAJ_PUSZCZENIE ; jeśli port P7 wciąż nie ma 0FH, klawisz jest trzymany
	
	MOV A, ZAPIS_KLAW	; zwróć bezpiecznie zapamiętany klawisz
	RET

; --- PODPROGRAM WPISYWANIA LICZBY DWUCYFROWEJ ---
WPROWADZ_WARTOSC:
	MOV BUFOR_CYFR,#0	
	CLR F0				; F0=0 -> czekamy na cyfrę dziesiątek
CZEKAJ_CYFRA:
	ACALL CZEKAJ_NA_KLAWISZ
	CJNE A,#11,DALEJ_SPRAWDZAJ ; Jeśli naciśnięty klawisz to '#', kończymy
	
	; Zatwierdzenie klawiszem '#' -> wyjdź i zwróć wartość
	MOV A, BUFOR_CYFR
	RET

DALEJ_SPRAWDZAJ:
	MOV ZAPIS_KLAW, A	
	CLR C
	SUBB A,#10
	JNC CZEKAJ_CYFRA	; Jeśli wciśnięto '*' lub błędny kod funkcyjny -> zignoruj

	; Sprawdź, czy wpisujemy cyfrę dziesiątek czy jedności
	JB F0, WPISZ_JEDNOSC
	
	; 1. Pierwsza cyfra (dziesiątki)
	MOV A, ZAPIS_KLAW
	MOV B,#10
	MUL AB				; pomnóż przez 10 (np. 2 * 10 = 20)
	MOV BUFOR_CYFR, A
	SETB F0				; następny klawisz wejdzie jako jedność
	SJMP ODASWIEZ_EKRAN

WPISZ_JEDNOSC:
	; 2. Druga cyfra (jedności)
	MOV A, BUFOR_CYFR
	ADD A, ZAPIS_KLAW	; dodaj nową cyfrę do dziesiątek
	
	; Kontrola poprawnego zakresu wpisanej liczby
	MOV R2, A			
	CLR C
	SUBB A, LIMIT_WPR	; porównaj z limitem (24 lub 60)
	JNC ZLY_ZAKRES		; jeśli wpisana liczba jest większa/równa limitowi -> błąd!
	
	MOV A, R2        
	MOV BUFOR_CYFR, A	; liczba poprawna, zapisz w buforze
	CLR F0				; zresetuj flage, aby móc w razie czego pisać od nowa przed kliknięciem #
	SJMP ODASWIEZ_EKRAN

ZLY_ZAKRES:
	MOV BUFOR_CYFR, #0	; przekroczono zakres -> wyzeruj pozycję
	CLR F0

ODASWIEZ_EKRAN:
	; Przypisz wartość w locie do rejestrów, aby na LCD od razu było widać zmiany
	MOV A, LIMIT_WPR
	CJNE A,#24,ZAPISZ_MINUTY
	MOV A, BUFOR_CYFR        
	MOV R5, A			; aktualizuj godziny na ekranie
	SJMP RYSUJ
ZAPISZ_MINUTY:
	MOV A, BUFOR_CYFR        
	MOV R6, A			; aktualizuj minuty na ekranie
RYSUJ:
	ACALL POKAZ_ZEGAR	
	LJMP CZEKAJ_CYFRA

; =========================================================================
; GŁÓWNY START PROGRAMU
; =========================================================================

START:
	init_LCD
	
	; Wyzerowanie rejestrów czasu na ekranie startowym
	MOV R5,#00H	
	MOV R6,#00H	
	MOV R7,#00H	
	ACALL POKAZ_ZEGAR

	; --- BLOK KONFIGURACJI PRZED STARTEM (Zatwierdzany przez #) ---
	; 1. Ustawianie Godzin (zakres do 24)
	MOV LIMIT_WPR, #24
	ACALL WPROWADZ_WARTOSC
	MOV R5, A			
	
	; 2. Ustawianie Minut (zakres do 60)
	MOV LIMIT_WPR, #60
	ACALL WPROWADZ_WARTOSC
	MOV R6, A			

	; --- INICJALIZACJA HARDWARE TIMERA ---
	MOV TMOD,#01H 		
	MOV TH0,#3CH 		
	MOV TL0,#0B0H 		
	SETB TR0      		
	MOV IE,#82H  		
	
	MOV R4,#01H			; Domyślnie zegar startuje od razu po wpisaniu danych
	MOV R0,#20 			

	ACALL POKAZ_ZEGAR

; Pętla główna urządzenia (Monitorowanie przycisków zewnętrznych i czasu)
MAIN:
	; --- GUZIK STOP (Pin P3.2) ---
	JB P3.2, START_BTN
	MOV R4,#00H			; Zatrzymanie odliczania zegara
WAIT_STOP:
	JNB P3.2, WAIT_STOP	

START_BTN:
	; --- GUZIK START (Pin P3.3) ---
	JB P3.3, PLUS_BTN
	MOV R0,#20			; Reset licznika czasu do równej pełnej sekundy
	MOV R4,#01H			; Uruchomienie odliczania zegara
WAIT_START:
	JNB P3.3, WAIT_START

PLUS_BTN:
	JB P3.4, MINUS_BTN
	ACALL GODZINA_PLUS
WAIT1:
	JNB P3.4, WAIT1		

MINUS_BTN:
	JB P3.5, LICZNIK
	ACALL GODZINA_MINUS
WAIT2:
	JNB P3.5, WAIT2		

; Sekcja asynchronicznego odliczania czasu
LICZNIK:
	MOV A,R4		; sprawdź flagę pracy
	JZ MAIN			; jeśli status = STOP, ignoruj timer i wracaj na początek pętli MAIN

	MOV A,R0   		; czy minęła 1 sekunda (20 przerwań timera)?
	JNZ MAIN   		; jeśli nie, idź dalej badać przyciski w MAIN

	MOV R0,#20		; przeładuj licznik systemowy na nową sekundę
	ACALL ZEGAR		
	LJMP MAIN		

; =========================================================================
; TABELA KODÓW KLAWIATURY MATRYCOWEJ DLA MIKROKOMPUTERA ZD537
; =========================================================================
TAB_KLAWISZ: 
	DB 077h, 07Bh, 07Dh, 0B7h, 0BBh, 0BDh, 0D7h, 0DBh, 0DDh, 0E7h, 0EBh, 0EDh

END START
