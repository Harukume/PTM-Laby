ljmp start

P5 equ 0F8H
P7 equ 0DBH
	
LCDstatus  equ 0FF2EH       ; adres do odczytu gotowosci LCD
LCDcontrol equ 0FF2CH       ; adres do podania bajtu sterujacego LCD
LCDdataWR  equ 0FF2DH       ; adres do podania kodu ASCII na LCD

// bajty sterujace LCD
#define  HOME     0x80     
#define  INITDISP 0x38     
#define  HOM2     0xc0     
#define  LCDON    0x0e     
#define  CLEAR    0x01     

// linie klawiatury - sterowanie na port P5
#define LINE_1		0x7f	// 0111 1111
#define LINE_2		0xbf	// 1011 1111
#define	LINE_3		0xdf	// 1101 1111
#define LINE_4		0xef	// 1110 1111

; --- ALOKACJA ZMIENNYCH W PAMIĘCI RAM ---
LIMIT_WPR   EQU 30H   ; Przechowuje aktualny limit (24 dla godzin, 60 dla minut)
BUFOR_CYFR  EQU 31H   ; Rejestr budowanej liczby dwucyfrowej
KOD_KLAW    EQU 32H   ; Przechowuje zdekodowany kod klawisza

ORG 000BH     				; obsluga przerwania Timera 0
	MOV TH0, #3CH 			; przeladowanie na 50ms
	MOV TL0, #0B0H 			
	DEC R0        			; korekta licznika programowego (20 * 50ms = 1s)
	RETI          			

org 0100H
		
// makra do obsługi wyświetlacza LCD
LCDcntrlWR MACRO x          
           LOCAL loop       
loop: MOV  DPTR,#LCDstatus  
      MOVX A,@DPTR          
      JB   ACC.7,loop       
      MOV  DPTR,#LCDcontrol 
      MOV  A, x             
      MOVX @DPTR,A          
      ENDM
	  
LCDcharWR MACRO
      LOCAL tutu            
      PUSH ACC              
tutu: MOV  DPTR,#LCDstatus  
      MOVX A,@DPTR          
      JB   ACC.7,tutu       
      MOV  DPTR,#LCDdataWR  
      POP  ACC              
      MOVX @DPTR,A          
      ENDM
	  
init_LCD MACRO
         LCDcntrlWR #INITDISP 
         LCDcntrlWR #CLEAR    
         LCDcntrlWR #LCDON    
         ENDM
		 
// funkcja wypisania liczby dwucyfrowej dla potrzeb zegara (np. 08, 12, 23)
putdigitLCD:	mov b, #10
				div ab				; cyfra dziesiątek w A, jedności w B
				add a, #30H			; konwersja na ASCII
				acall putcharLCD
				mov a, b			
				add a, #30H			
				acall putcharLCD
				ret

putcharLCD:	LCDcharWR
			ret

; wyswietlanie czasu
POKAZ_ZEGAR:
	LCDcntrlWR #HOME	
	MOV A,R5		; godziny
	ACALL putdigitLCD
	MOV A,#":"		
	ACALL putcharLCD
	MOV A,R6		; minuty
	ACALL putdigitLCD
	MOV A,#":"		
	ACALL putcharLCD
	MOV A,R7		; sekundy
	ACALL putdigitLCD
	RET

; aktualizacja czasu w tle
ZEGAR:
	INC R7		; sekundy++
	MOV A,R7
	CLR C
	SUBB A,#60	
	JZ MINUTY
	ACALL POKAZ_ZEGAR
	RET

MINUTY:
	MOV R7,#00H	
	INC R6		; minuty++
	MOV A,R6
	CLR C
	SUBB A,#60	
	JZ GODZINY
	ACALL POKAZ_ZEGAR
	RET

GODZINY:
	MOV R6,#00H	
	INC R5		; godziny++
	MOV A,R5
	CLR C
	SUBB A,#24	
	JNZ EKRAN
	MOV R5,#00H	; nowa doba
EKRAN:
	ACALL POKAZ_ZEGAR
	RET

; dodawanie godziny przyciskiem PLUS
GODZINA_PLUS:
	INC R5		
	MOV A,R5
	CLR C
	SUBB A,#24	
	JNZ PLUS_OK
	MOV R5,#00H	
PLUS_OK:
	ACALL POKAZ_ZEGAR
	RET

; odejmowanie godziny przyciskiem MINUS
GODZINA_MINUS:
	MOV A,R5
	JNZ MINUS_NORMAL
	MOV R5,#23	
	SJMP MINUS_END
MINUS_NORMAL:
	DEC R5		
MINUS_END:
	ACALL POKAZ_ZEGAR
	RET

; =========================================================================
; PODPROGRAMY ODCZYTU KLAWIATURY MATRYCOWEJ
; =========================================================================

CZYTAJ_KLAWISZ:
	; Skanowanie linii 1
	MOV P5,#LINE_1
	MOV A,P7
	ANL A,#0FH
	XRL A,#0FH
	JNZ KLAW_OK
	; Skanowanie linii 2
	MOV P5,#LINE_2
	MOV A,P7
	ANL A,#0FH
	XRL A,#0FH
	JNZ KLAW_OK
	; Skanowanie linii 3
	MOV P5,#LINE_3
	MOV A,P7
	ANL A,#0FH
	XRL A,#0FH
	JNZ KLAW_OK
	; Skanowanie linii 4
	MOV P5,#LINE_4
	MOV A,P7
	ANL A,#0FH
	XRL A,#0FH
	JNZ KLAW_OK
	
	MOV A,#0FFH			; brak wciśniętego klawisza
	RET

KLAW_OK:
	MOV R1,P7			; zapisz kolumny
	MOV A,P5			; zapisz linie
	ANL A,#0F0H
	MOV R2,A			; tymczasowo linie w R2
	MOV A,R1
	ANL A,#0FH
	ORL A,R2
	MOV R1,A			; R1 = pełny unikalny kod klawisza (linia + kolumna)

	; Przeszukiwanie tabeli znaków
	MOV DPTR,#TAB_KLAWISZ
	MOV R2,#0
SZUKAJ_KODU:
	MOV A,R2
	MOVC A,@A+DPTR
	XRL A,R1
	JZ ZNALEZIONO
	INC R2
	CJNE R2,#12,SZUKAJ_KODU
	MOV A,#0FFH
	RET
ZNALEZIONO:
	MOV A,R2			; A zwraca: 0-9 = cyfry, 10 = *, 11 = #
	RET

CZEKAJ_NA_KLAWISZ:
	; Pętla 1: Czekaj, aż jakikolwiek klawisz zostanie wciśnięty
	ACALL CZYTAJ_KLAWISZ
	XRL A,#0FFH
	JZ CZEKAJ_NA_KLAWISZ
	
	MOV KOD_KLAW, A		; Zapisz kod wciśniętego klawisza
	
	; Pętla 2: Czekaj stabilnie, aż użytkownik PUŚCI klawisz (eliminacja wielokrotnego odczytu)
CZEKAJ_PUSZCZENIE:
	ACALL CZYTAJ_KLAWISZ
	XRL A,#0FFH
	JNZ CZEKAJ_PUSZCZENIE
	
	MOV A, KOD_KLAW		; Zwróć zapisany kod w akumulatorze A
	RET

; --- UNIWERSALNA PĘTLA BUDOWANIA LICZBY DWUCYFROWEJ ---
WPROWADZ_WARTOSC:
	MOV BUFOR_CYFR,#0	; Zeruj budowaną liczbę
	CLR F0				; F0 = 0 -> oczekiwanie na pierwszą cyfrę (dziesiątek)
CZEKAJ_CYFRA:
	ACALL CZEKAJ_NA_KLAWISZ
	CJNE A,#11,SPRAWDZ_CZY_CYFRA ; Jeśli to nie '#', sprawdź czy cyfra
	
	; Wciśnięto '#' -> Koniec wprowadzania tej pozycji, zatwierdź i wyjdź
	MOV A, BUFOR_CYFR
	RET

SPRAWDZ_CZY_CYFRA:
	MOV KOD_KLAW, A		; Przechowaj kod klawisza
	CLR C
	SUBB A,#10
	JNC CZEKAJ_CYFRA	; Jeśli wciśnięto '*' (kod 10) lub inny błąd -> ignoruj, czekaj dalej
	
	; Decyzja: czy to dziesiątki czy jedności
	JB F0,DODAJ_JEDNOSC
	
	; 1. Pozycja dziesiątek
	MOV A, KOD_KLAW
	MOV B,#10
	MUL AB				; Cyfra * 10
	MOV BUFOR_CYFR, A
	SETB F0				; Flaga na 1 -> następna cyfra wejdzie jako jedność
	SJMP AKTUALIZUJ_WIDOK

DODAJ_JEDNOSC:
	; 2. Pozycja jedności
	MOV A, BUFOR_CYFR
	ADD A, KOD_KLAW		; Stara wartość dziesiątek + nowa cyfra jedności
	
	; --- KONTROLA ZAKRESU (Czy uzyskana liczba < LIMIT_WPR?) ---
	MOV R2, A			; Schowaj potencjalny wynik do R2
	CLR C
	SUBB A, LIMIT_WPR	; Odejmij limit (24 lub 60)
	JNC BLAD_ZAKRESU	; Jeśli wynik >= limitu, zgłoś błąd i wyzeruj pozycję
	
	MOV A, R2        
	MOV BUFOR_CYFR, A	; Wynik jest w porządku, zapisz go w buforze
	CLR F0				; Zresetuj flagę (pozwala pisać od nowa, jeśli użytkownik zmieni zdanie przed '#')
	SJMP AKTUALIZUJ_WIDOK

BLAD_ZAKRESU:
	MOV BUFOR_CYFR, #0	; Resetuj bufor do 00 przy przekroczeniu zakresu
	CLR F0				; Pisz od nowa od dziesiątek

AKTUALIZUJ_WIDOK:
	; W zależności od ustawianego parametru uaktualnij odpowiedni rejestr czasu w locie
	MOV A, LIMIT_WPR
	CJNE A,#24,UST_MINUTY
	MOV A, BUFOR_CYFR        
	MOV R5, A			; Podgląd godzin na LCD
	SJMP RYSUJ
UST_MINUTY:
	MOV A, BUFOR_CYFR        
	MOV R6, A			; Podgląd minut na LCD
RYSUJ:
	ACALL POKAZ_ZEGAR	; Odświeżenie ekranu
	LJMP CZEKAJ_CYFRA

; =========================================================================
; GŁÓWNY PROGRAM
; =========================================================================

START:
	init_LCD
	
	; Wstępne wyczyszczenie rejestrów czasu przed konfiguracją
	MOV R5,#00H	
	MOV R6,#00H	
	MOV R7,#00H	
	ACALL POKAZ_ZEGAR

	; --- SEKCJA USTAWIANIA CZASU PRZED URUCHOMIENIEM ---
	; 1. Konfiguracja Godzin (zakres 00 - 23)
	MOV LIMIT_WPR, #24
	ACALL WPROWADZ_WARTOSC
	MOV R5, A			; Zapisz ostatecznie zatwierdzone godziny w R5
	
	; 2. Konfiguracja Minut (zakres 00 - 59)
	MOV LIMIT_WPR, #60
	ACALL WPROWADZ_WARTOSC
	MOV R6, A			; Zapisz ostatecznie zatwierdzone minuty w R6

	; --- START TIMERA (Dopiero po konfiguracji klawiaturą) ---
	MOV TMOD,#01H 		
	MOV TH0,#3CH 		
	MOV TL0,#0B0H 		
	SETB TR0      		
	MOV IE,#82H  		
	
	MOV R4,#01H			; Uruchomienie odliczania zegara (01H-start)
	MOV R0,#20 			; Licznik bazowy dla 1 sekundy

	ACALL POKAZ_ZEGAR

; Pętla główna (Obsługa przycisków zewnętrznych P3)
MAIN:
	; --- GUZIK P3.2: STOP ---
	JB P3.2, START_BTN
	MOV R4,#00H			; Zatrzymanie logiczne odliczania
WAIT_STOP:
	JNB P3.2, WAIT_STOP	; Czekaj na puszczenie przycisku STOP

START_BTN:
	; --- GUZIK P3.3: START ---
	JB P3.3, PLUS_BTN
	MOV R0,#20			; Reset licznika programowego do pełnej sekundy przy wznowieniu
	MOV R4,#01H			; Uruchomienie logiczne odliczania
WAIT_START:
	JNB P3.3, WAIT_START; Czekaj na puszczenie przycisku START

PLUS_BTN:
	JB P3.4, MINUS_BTN
	ACALL GODZINA_PLUS
WAIT1:
	JNB P3.4, WAIT1		

MINUS_BTN:
	JB P3.5, LICZNIK
	ACALL GODZINA_MINUS
WAIT2:
	JNB P3.5, WAIT2		

; Sekcja nieblokującego odmierzania 1 sekundy
LICZNIK:
	MOV A,R4		; Sprawdź flagę start/stop
	JZ MAIN			; Jeśli STOP (R4=0) -> pomiń krok timera, skocz do pętli MAIN

	MOV A,R0   		; Sprawdź czy minęło 20 przerwań (1 sekunda)
	JNZ MAIN   		; Jeśli sekunda jeszcze nie upłynęła, wróć sprawdzać przyciski

	MOV R0,#20		; Przeładuj licznik programowy na nową sekundę
	ACALL ZEGAR		; Wywołaj podprogram aktualizacji czasu
	LJMP MAIN		

; =========================================================================
; TABELA MAPOWANIA KLAWIATURY MATRYCOWEJ
; =========================================================================
TAB_KLAWISZ: 
	DB 077h, 07Bh, 07Dh, 0B7h, 0BBh, 0BDh, 0D7h, 0DBh, 0DDh, 0E7h, 0EBh, 0EDh

END START
