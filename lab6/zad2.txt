ljmp start

P5 equ 0F8H
P7 equ 0DBH
	
LCDstatus  equ 0FF2EH       ; adres do odczytu gotowosci LCD
LCDcontrol equ 0FF2CH       ; adres do podania bajtu sterujacego LCD
LCDdataWR  equ 0FF2DH       ; adres do podania kodu ASCII na LCD

// bajty sterujace LCD
#define  HOME     0x80     
#define  INITDISP 0x38     
#define  HOM2     0xc0     
#define  LCDON    0x0e     
#define  CLEAR    0x01     

// linie klawiatury - sterowanie na port P5
#define LINE_1		0x7f	// 0111 1111
#define LINE_2		0xbf	// 1011 1111
#define	LINE_3		0xdf	// 1101 1111
#define LINE_4		0xef	// 1110 1111
#define ALL_LINES	0x0f	// 0000 1111

; --- REJESTRY PAMIĘCI RAM DO BUDOWANIA LICZBY ---
LIMIT_WPR   EQU 30H   ; Przechowuje aktualny limit (24 dla godzin, 60 dla minut)
BUFOR_CYFR  EQU 31H   ; Tu budujemy liczbę dwucyfrową
KOD_KLAWISZA EQU 32H  ; Przechowalnia tymczasowa zdekodowanego klawisza

ORG 000BH     				; obsluga przerwania
	MOV TH0, #3CH 			; przeladowanie
	MOV TL0, #0B0H 			; stalej timera na 50ms
	DEC R0        			; korekta licznika
	RETI          			; powrot z przerwania

org 0100H
		
// macro do wprowadzenia bajtu sterujacego na LCD
LCDcntrlWR MACRO x          
           LOCAL loop       
loop: MOV  DPTR,#LCDstatus  
      MOVX A,@DPTR          
      JB   ACC.7,loop       
      MOV  DPTR,#LCDcontrol 
      MOV  A, x             
      MOVX @DPTR,A          
      ENDM
	  
// macro do wypisania znaku ASCII na LCD
LCDcharWR MACRO
      LOCAL tutu            
      PUSH ACC              
tutu: MOV  DPTR,#LCDstatus  
      MOVX A,@DPTR          
      JB   ACC.7,tutu       
      MOV  DPTR,#LCDdataWR  
      POP  ACC              
      MOVX @DPTR,A          
      ENDM
	  
// macro do inicjalizacji wyswietlacza
init_LCD MACRO
         LCDcntrlWR #INITDISP 
         LCDcntrlWR #CLEAR    
         LCDcntrlWR #LCDON    
         ENDM

delay:	mov r1, #0FFH
dwa:	mov r2, #0FFH
trzy:	djnz r2, trzy
		djnz r1, dwa
		ret
		 
// funkcja wypisania liczby dla potrzeb zegara
putdigitLCD:	mov b, #10
				div ab				; uzyskanie cyfry dziesiatek
				add a, #30H			; konwersja cyfry na kod ASCII
				acall putcharLCD
				mov a, b			; ladowanie cyfry jednosci
				add a, #30H			; konwersja na LCD
				acall putcharLCD
				ret

// funkcja wypisywania znaku na LCD
putcharLCD:	LCDcharWR
			ret

// tablica przekodowania klawisze - ASCII w XRAM
keyascii_num:	mov dptr, #80EBH
			mov a, #0
			movx @dptr, a
			
			mov dptr, #8077H
			mov a, #1
			movx @dptr, a
			
			mov dptr, #807BH
			mov a, #2
			movx @dptr, a
			
			mov dptr, #807DH
			mov a, #3
			movx @dptr, a
			
			mov dptr, #80B7H
			mov a, #4
			movx @dptr, a
			
			mov dptr, #80BBH
			mov a, #5
			movx @dptr, a
			
			mov dptr, #80BDH
			mov a, #6
			movx @dptr, a
			
			mov dptr, #80D7H
			mov a, #7
			movx @dptr, a
			
			mov dptr, #80DBH
			mov a, #8
			movx @dptr, a
			
			mov dptr, #80DDH
			mov a, #9
			movx @dptr, a
			
			mov dptr, #807EH
			mov a, #"A"
			movx @dptr, a
			
			mov dptr, #80BEH
			mov a, #"B"
			movx @dptr, a
			
			mov dptr, #80DEH
			mov a, #"C"
			movx @dptr, a
			
			mov dptr, #80EEH
			mov a, #"D"
			movx @dptr, a
			
			mov dptr, #80E7H
			mov a, #"*"
			movx @dptr, a
			
			mov dptr, #80EDH
			mov a, #"#"
			movx @dptr, a
			ret

// odswiezanie i wyswietlanie aktualnego czasu na LCD
POKAZ_ZEGAR:
			LCDcntrlWR #HOME	
			MOV A, R5			; godziny
			ACALL putdigitLCD
			MOV A, #":"			
			ACALL putcharLCD
			MOV A, R6			; minuty
			ACALL putdigitLCD
			MOV A, #":"			
			ACALL putcharLCD
			MOV A, R7			; sekundy
			ACALL putdigitLCD
			RET

// wyznaczanie biezacej wartosci zegara
ZEGAR:		INC R7				; licznik sekund
			MOV A, R7			; obsluga sekund
			CLR C
			SUBB A, #60			; przepelnienie sekund
			JZ MINUTY
			ACALL POKAZ_ZEGAR
			JMP FINAL
MINUTY:		MOV R7, #00H		; zerowanie sekund
			INC R6				; licznik minut
			MOV A, R6			; obsluga minut
			CLR C
			SUBB A, #60			; przepelnienie minut
			JZ GODZINY
			ACALL POKAZ_ZEGAR
			JMP FINAL
GODZINY:	MOV R6, #00H		; zerowanie minut
			INC R5				; licznik godzin
			MOV A, R5
			CLR C
			SUBB A, #24			; przepelenienie godzin - doba
			JNZ EKRAN
			MOV R5, #00H		; zerowanie godzin
EKRAN:		ACALL POKAZ_ZEGAR
FINAL:		RET

; =========================================================================
; PODPROGRAMY AMORTYZACJI STYKÓW I ODCZYTU POJEDYNCZEGO KLIKNIĘCIA
; =========================================================================

CZYTAJ_MATRYCE:
		key_1:	mov r1, #LINE_1
				mov	a, r1
				mov	P5, a
				mov a, P7
				anl a, #0Fh		; ZD537 czyta dolną połówkę z P7
				cjne a, #0Fh, dekoduj
		key_2:	mov r1, #LINE_2
				mov	a, r1
				mov	P5, a
				mov a, P7
				anl a, #0Fh
				cjne a, #0Fh, dekoduj
		key_3:	mov r1, #LINE_3
				mov	a, r1
				mov	P5, a
				mov a, P7
				anl a, #0Fh
				cjne a, #0Fh, dekoduj
		key_4:	mov r1, #LINE_4
				mov	a, r1
				mov	P5, a
				mov a, P7
				anl a, #0Fh
				cjne a, #0Fh, dekoduj
				
				mov a, #0FFH	; brak wciśniętego klawisza
				ret

dekoduj:		mov r2, a		; zapas kolumn
				mov a, r1		; linia aktywująca
				anl a, #0F0H
				orl a, r2		; budujemy pełny adres dla XRAM
				mov dph, #80h
				mov dpl, a
				movx a, @dptr	; pobranie znaku z tabeli przekodowania (0-9, *, #, A-D)
				ret

POBIERZ_KLIK:
				acall CZYTAJ_MATRYCE
				cjne a, #0FFH, klawisz_jest
				sjmp POBIERZ_KLIK	; czekamy aż użytkownik coś naciśnie
klawisz_jest:	mov KOD_KLAWISZA, a ; zapisujemy co wcisnął

puszczenie:		mov P5, #00H		; odcinamy linie
				mov a, P7
				anl a, #0Fh
				cjne a, #0Fh, puszczenie ; pętla czeka, aż fizycznie puścisz klawisz
				
				acall delay			; programowe usuwanie drgań zestyków
				mov a, KOD_KLAWISZA
				ret

; =========================================================================
; UNIWERSALNA PROCEDURA BUDOWANIA LICZBY DWUCYFROWEJ Z KONTROLĄ
; =========================================================================
WPROWADZ_WARTOSC:
				MOV BUFOR_CYFR, #0	; zerujemy bufor na start
				CLR F0				; F0=0 -> oczekujemy na pierwszą cyfrę (dziesiątek)
czekaj_na_cyfre:
				acall POBIERZ_KLIK
				cjne a, #"#", sprawdz_czy_cyfra
				
				; Naciśnięto '#' -> Kończymy wprowadzanie, aktualny stan bufora to nasz wynik
				mov a, BUFOR_CYFR
				ret

sprawdz_czy_cyfra:
				; Sprawdzamy czy to znak ASCII z przedziału '0'-'9' (czyli 0x00 do 0x09 w Twojej tabeli)
				clr c
				subb a, #10
				jnc czekaj_na_cyfre ; jeśli naciśnięto klawisze funkcyjne (*, A, B...) -> ignoruj

				jb F0, wpisz_jedn
				
				; KROK 1: Pierwsza cyfra (Dziesiątki)
				mov a, KOD_KLAWISZA
				mov b, #10
				mul ab				; np. wpisano 1 -> 1 * 10 = 10
				mov BUFOR_CYFR, a
				setb F0				; flaga na 1 -> następna cyfra wpadnie do jedności
				sjmp odswiez_konfiguracje

wpisz_jedn:
				; KROK 2: Druga cyfra (Jedności)
				mov a, BUFOR_CYFR
				add a, KOD_KLAWISZA ; dodajemy cyfrę jedności do dziesiątek
				
				; Kontrola zakresu (czy zbudowana liczba < LIMIT_WPR?)
				mov r2, a			; zapas
				clr c
				subb a, LIMIT_WPR
				jnc blad_limitu		; jeśli wynik >= limitu -> błąd!
				
				mov a, r2
				mov BUFOR_CYFR, a	; wartość poprawna, zapisz
				clr F0				; reset flagi (można poprawiać liczbę na nowo, dopóki nie kliknie się #)
				sjmp odswiez_konfiguracje

blad_limitu:
				mov BUFOR_CYFR, #0	; czyszczenie przy błędzie zakresu
				clr F0

odswiez_konfiguracje:
				; Podgląd wpisywanej wartości w czasie rzeczywistym na ekranie
				mov a, LIMIT_WPR
				cjne a, #24, podglad_minut
				mov a, BUFOR_CYFR
				mov r5, a			; chwilowo podmieniamy godziny w R5
				sjmp rysuj_ekran
podglad_minut:
				mov a, BUFOR_CYFR
				mov r6, a			; chwilowo podmieniamy minuty w R6
rysuj_ekran:
				acall POKAZ_ZEGAR
				ljmp czekaj_na_cyfre

; =========================================================================
; GŁÓWNY PROGRAM
; =========================================================================

START:	init_LCD
		acall keyascii_num
		
		; Inicjalizacja ekranu startowego (czyste zera)
		MOV R5, #00H			
		MOV R6, #00H
		MOV R7, #00H
		ACALL POKAZ_ZEGAR		

		; --- ETAP USTAWIANIA GODZIN (Limit: 24) ---
		MOV LIMIT_WPR, #24
		ACALL WPROWADZ_WARTOSC
		MOV R5, A				; ostatecznie zatwierdzona liczba trafia do R5

		; --- ETAP USTAWIANIA MINUT (Limit: 60) ---
		MOV LIMIT_WPR, #60
		ACALL WPROWADZ_WARTOSC
		MOV R6, A				; ostatecznie zatwierdzona liczba trafia do R6

		; --- URUCHOMIENIE SPRZĘTOWEGO ODMIERZANIA CZASU (Dopiero po ustawieniu!) ---
		MOV TMOD, #01H 			; konfiguracja timera
		MOV TH0, #3CH 			; ladowanie
		MOV TL0, #0B0H 			; stalej timera na 50ms
		SETB TR0	 			; timer start
		MOV IE, #82H  			; przerwania wlacz
		MOV R7, #00H			; start od 00 sekund
		MOV R0, #20 			; licznik odmierzen 20 x 50ms = 1 sekunda
		
		MOV A, #0FH
		MOV P1, A 				; zapalenie diod
		ACALL POKAZ_ZEGAR

CZEKAM: 		
		; Stałe i nieblokujące monitorowanie wejść klawiatury podczas pracy zegara
		; Klawisze A (START), B (STOP), C (RESET) zgodnie z Twoją logiką
		acall CZYTAJ_MATRYCE
		cjne a, #"A", sprawdz_B
		SETB TR0				; wznowienie odliczania
		sjmp skip_klaw
sprawdz_B:
		cjne a, #"B", sprawdz_C
		clr TR0					; zatrzymanie odliczania
		sjmp skip_klaw
sprawdz_C:
		cjne a, #"C", skip_klaw
		ljmp START				; całkowity restart i powrót do konfiguracji
		
skip_klaw:
		MOV A, R0    			; sprawdzenie licznika sekund programowych
		JNZ CZEKAM    			
		
		MOV R0, #20				; przeładowanie bazy 1 sekundy
		ACALL ZEGAR				; aktualizacja czasu
		
		MOV A, P1 				; mruganie diodami kontrolnymi
		CPL A       			
		MOV P1, A
		
		JMP CZEKAM    			

END START
