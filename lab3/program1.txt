ljmp start

; --- ADRESY LCD ---
LCDstatus  equ 0FF2EH       
LCDcontrol equ 0FF2CH       
LCDdataWR  equ 0FF2DH       

; --- KOMENDY STERUJĄCE ---
#define  CLEAR    0x01
#define  INITDISP 0x38
#define  LCDON    0x0e
#define  HOME     0x80

org 0100H

; --- DEKLARACJE TEKSTÓW ---
text1:  db "Guzik 1", 0
text2:  db "Guzik 2", 0
text3:  db "Guzik 3", 0
text4:  db "Guzik 4", 0
msg_ex: db "KONIEC", 0

; --- TWOJE SPRAWDZONE MAKRA ---
LCDcntrlWR MACRO x
    LOCAL loop
loop: MOV  DPTR,#LCDstatus
      MOVX A,@DPTR
      JB   ACC.7,loop
      MOV  DPTR,#LCDcontrol
      MOV  A, x
      MOVX @DPTR,A
      ENDM
      
LCDcharWR MACRO
      LOCAL tutu
      PUSH ACC
tutu: MOV  DPTR,#LCDstatus
      MOVX A,@DPTR
      JB   ACC.7,tutu
      MOV  DPTR,#LCDdataWR
      POP  ACC
      MOVX @DPTR,A
      ENDM

init_LCD MACRO
         LCDcntrlWR #INITDISP
         LCDcntrlWR #CLEAR
         LCDcntrlWR #LCDON
         ENDM

; --- FUNKCJE POMOCNICZE ---
delay:  mov r0, #15H
one:    mov r1, #0FFH
dwa:    mov r2, #0FFH
trzy:   djnz r2, trzy
        djnz r1, dwa
        djnz r0, one
        ret

; Poprawiona funkcja wypisywania ciągu znaków
putstrLCD:  
    clr a
    movc a, @a+dptr     ; Pobierz literę
    jz koniec           ; Jeśli 0, wyjdź
    
    ; KLUCZOWA POPRAWKA: Ratujemy DPTR przed makrem
    push dph
    push dpl
    
    LCDcharWR           ; Wypisz literę (to makro niszczy DPTR)
    
    pop dpl             ; Przywracamy DPTR z tekstem
    pop dph
    
    inc dptr            ; Teraz możemy bezpiecznie przejść do kolejnej litery
    sjmp putstrLCD
koniec: 
    ret

; --- OBSŁUGA PRZYCISKÓW ---
przycisk1:
    LCDcntrlWR #CLEAR
    mov dptr, #text1
    acall putstrLCD
    acall delay
    ljmp begin

przycisk2:
    LCDcntrlWR #CLEAR
    mov dptr, #text2
    acall putstrLCD
    acall delay
    ljmp begin

przycisk3:
    LCDcntrlWR #CLEAR
    mov dptr, #text3
    acall putstrLCD
    acall delay
    ljmp begin

przycisk4:
    LCDcntrlWR #CLEAR
    mov dptr, #text4
    acall putstrLCD
    acall delay
    ljmp begin

; --- PROGRAM GŁÓWNY ---
start:  
    mov sp, #30h        ; Ustawienie stosu
    init_LCD            ; Inicjalizacja LCD

begin:  
    clr c
    mov a, p3
    subb a, #0cfh       ; Koniec (oba skrajne)
    jz finito

    mov a, p3
    subb a, #0dfh       ; Przycisk 1
    jz skok1

    mov a, p3
    subb a, #0efh       ; Przycisk 2
    jz skok2

    mov a, p3
    subb a, #0f7h       ; Przycisk 3
    jz skok3

    mov a, p3
    subb a, #0fbh       ; Przycisk 4
    jz skok4

    sjmp begin          ; Nic nie wciśnięto - zapętl

skok1: ljmp przycisk1
skok2: ljmp przycisk2
skok3: ljmp przycisk3
skok4: ljmp przycisk4

finito: 
    LCDcntrlWR #CLEAR
    mov dptr, #msg_ex
    acall putstrLCD
    sjmp $

end start
