; Adresy LCD (zestaw ZD537)
LCDstatus   equ 0FF2EH 
LCDcontrol  equ 0FF2CH 
LCDdataWR   equ 0FF2DH 
P3          EQU 0B0h

; Instrukcje sterujące
#define CLEAR    0x01
#define INITDISP 0x38
#define LCDON    0x0e

; --- MAKRA ---
LCDcntrlWR MACRO x 
    LOCAL loop 
loop:
    MOV  DPTR,#LCDstatus 
    MOVX A,@DPTR 
    JB   ACC.7,loop 
    MOV  DPTR,#LCDcontrol 
    MOV  A, x 
    MOVX @DPTR,A
ENDM

LCDcharWR MACRO
    LOCAL loop1
    PUSH ACC 
loop1: 
    MOV  DPTR,#LCDstatus
    MOVX A,@DPTR
    JB   ACC.7,loop1 
    POP  ACC 
    MOV  DPTR,#LCDdataWR 
    MOVX @DPTR,A
ENDM

init_LCD MACRO
    LCDcntrlWR #INITDISP
    LCDcntrlWR #CLEAR
    LCDcntrlWR #LCDON
ENDM

; --- PROGRAM GŁÓWNY ---
ORG 8000h
start:
    mov sp, #30h
    init_LCD

main_loop:
    mov a, P3           
    cpl a               ; Odwróć (1 = wciśnięty)
    anl a, #00111100b   ; MASKA: zostaw tylko bity 5, 4, 3, 2
    
    mov r2, a           ; Zapamiętaj stan w R2
    
    ; Sprawdzenie wyjścia (oba skrajne: bit 2 i bit 5 -> 00100100b)
    anl a, #00100100b   
    cjne a, #00100100b, check_buttons 
    
    ; Wyjście (Guzik 2 i 5 wciśnięty)
    LCDcntrlWR #CLEAR
    mov dptr, #msg_exit
    acall print_string
    sjmp finish

check_buttons:
    mov a, r2           ; Przywróć stan guzików
    jz main_loop        

    ; Wybór przycisku (zgodnie z Twoim nowym układem pinów)
    jb acc.2, btn1      ; Pierwszy guzik jest na bicie 2
    jb acc.3, btn2      ; Drugi na 3
    jb acc.4, btn3      ; Trzeci na 4
    jb acc.5, btn4      ; Czwarty na 5
    sjmp main_loop

btn1:
    mov dptr, #txt1
    sjmp display
btn2:
    mov dptr, #txt2
    sjmp display
btn3:
    mov dptr, #txt3
    sjmp display
btn4:
    mov dptr, #txt4
    sjmp display

display:
    LCDcntrlWR #CLEAR   
    acall print_string
    acall delay         
    sjmp main_loop

finish:
    sjmp $

; --- PODPROGRAMY ---

print_string:
    push acc
ps_loop:
    clr  a
    movc a, @a+dptr     
    jz   ps_end         
    
    push dph            ; Ochrona DPTR przed makrem
    push dpl
    LCDcharWR           
    pop  dpl
    pop  dph
    
    inc  dptr
    sjmp ps_loop
ps_end:
    pop  acc
    ret

delay:
    mov r5, #5
del_outer:
    mov r6, #100
d1: mov r7, #250
d2: djnz r7, d2
    djnz r6, d1
    djnz r5, del_outer
    ret

; --- DANE ---
txt1: db "Przycisk 1 (P3.2)", 0
txt2: db "Opcja 2 (P3.3)", 0
txt3: db "Wybrano 3 (P3.4)", 0
txt4: db "Koniec (P3.5)", 0
msg_exit: db "KONIEC PROGRAMU", 0

END
