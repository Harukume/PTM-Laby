; Adresy LCD (zestaw ZD537)
LCDstatus   equ 0FF2EH
LCDcontrol  equ 0FF2CH
LCDdataWR   equ 0FF2DH

; Komendy sterujące
#define CLEAR    0x01
#define INITDISP 0x38
#define LCDON    0x0e
#define LINE1    0x80 
#define LINE2    0xc0 

; --- MAKRA ---
LCDcntrlWR MACRO x
    LOCAL loop
loop:
    MOV  DPTR,#LCDstatus
    MOVX A,@DPTR
    JB   ACC.7,loop
    MOV  DPTR,#LCDcontrol
    MOV  A, x
    MOVX @DPTR,A
ENDM

LCDcharWR MACRO
    LOCAL loop1
    PUSH ACC
loop1: 
    MOV  DPTR,#LCDstatus
    MOVX A,@DPTR
    JB   ACC.7,loop1
    POP  ACC
    MOV  DPTR,#LCDdataWR
    MOVX @DPTR,A
ENDM

init_LCD MACRO
    LCDcntrlWR #INITDISP
    LCDcntrlWR #CLEAR
    LCDcntrlWR #LCDON
ENDM

; --- PROGRAM GŁÓWNY ---
ORG 8000h
start:
    mov sp, #30h
    init_LCD
    mov dptr, #long_text 

next_block:
    ; Czyścimy ekran przed każdą nową "stroną" (32 znaki)
    push dph
    push dpl
    LCDcntrlWR #CLEAR   
    pop  dpl
    pop  dph
    
    ; --- LINIA 1 ---
    push dph
    push dpl
    LCDcntrlWR #LINE1
    pop  dpl
    pop  dph

    mov r2, #16         
line1_loop:
    clr  a
    movc a, @a+dptr
    jz   end_of_text    
    
    push dph            ; Ratuj DPTR przed makrem znaku
    push dpl
    LCDcharWR
    pop  dpl
    pop  dph
    
    inc  dptr
    djnz r2, line1_loop 

    ; --- LINIA 2 ---
    push dph            ; Ratuj DPTR przed makrem zmiany linii!
    push dpl
    LCDcntrlWR #LINE2
    pop  dpl
    pop  dph

    mov r2, #16         
line2_loop:
    clr  a
    movc a, @a+dptr
    jz   end_of_text    
    
    push dph            ; Ratuj DPTR
    push dpl
    LCDcharWR
    pop  dpl
    pop  dph
    
    inc  dptr
    djnz r2, line2_loop 

    ; --- PAUZA ---
    acall long_delay
    sjmp next_block     

end_of_text:
    acall long_delay
    LCDcntrlWR #CLEAR
    sjmp $              

; --- DELAY ---
long_delay:             
    mov r5, #15
ld_loop:
    acall delay
    djnz r5, ld_loop
    ret

delay:            
    mov r6, #200
d1: mov r7, #250
d2: djnz r7, d2
    djnz r6, d1
    ret

; --- DANE ---
long_text: 
    db "To jest bardzo dlugi tekst, ktory nie zmiesci sie w jednej linii, "
    db "dlatego musimy go podzielic na bloki po szesnascie znakow. "
    db "Program automatycznie przelacza linie i czysci ekran.", 0

END
